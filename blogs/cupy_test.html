import matplotlib.pyplot as plt
array_lengths = np.arange(10000, 1000000, 1000)
plt.figure()
for array_length in array_lengths:
    start_time = time.time()
    x_cpu = np.random.randn(array_length)
    mid_time = time.time()
    x_gpu = cp.random.randn(array_length)
    end_time = time.time()
    plt.plot([array_length], [mid_time - start_time], '.', color = 'black')
    plt.plot([array_length], [end_time - mid_time], '.', color = 'red')
plt.show()
